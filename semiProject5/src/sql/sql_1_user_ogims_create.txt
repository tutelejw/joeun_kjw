
-- CMD 창 시작후 그냥 붙여넣으세요

-- 0. oracle 관리자 계정 접속 후 튜닝 및 사용자 삭제 초기화 (sysdba 접속해서 진행)
-- 관리자 권한으로 접속
sqlplus / as sysdba

-- 현재 설정 조회
SHOW PARAMETER processes;
SHOW PARAMETER sessions;
SHOW PARAMETER transactions;

-- 예시: 프로세스를 500으로 설정하고 관련 값 조정
ALTER SYSTEM SET processes = 500 SCOPE = SPFILE;
ALTER SYSTEM SET sessions = 555 SCOPE = SPFILE;
ALTER SYSTEM SET transactions = 610 SCOPE = SPFILE;

-- 데이터베이스 재시작 필요 - 시간 1분 ~ 5분 정도 걸림
SHUTDOWN IMMEDIATE;
STARTUP;


-- 1. 사용자 생성 (system/system 접속해서 진행)
DROP USER ogims CASCADE;
CREATE USER ogims IDENTIFIED BY "ogims";

-- 2. 기본 롤 부여 (CONNECT: 로그인용, RESOURCE: 테이블/시퀀스 등 객체 생성용)
GRANT CONNECT, RESOURCE TO ogims;

-- 3. 추가 권한 부여 (뷰, 프로시저, 트리거 등)
GRANT CREATE VIEW TO ogims;
GRANT CREATE PROCEDURE TO ogims;
GRANT CREATE TRIGGER TO ogims;

EXIT;

-- 4. ogims 계정 접속
sqlplus ogims/ogims
