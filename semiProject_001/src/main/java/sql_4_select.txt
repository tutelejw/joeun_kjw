SELECT volunteerid, TO_CHAR(createdat, 'YYYYMMDDHH24MI') AS created_fmt,
       TO_CHAR(starttime, 'YYYYMMDDHH24MI') AS start_fmt,
       TO_CHAR(endtime, 'YYYYMMDDHH24MI') AS end_fmt
FROM volunteer;


봉사요청(offer) 게시글 목록 조회
 SELECT 
  VOLUNTEERID,
  '(' || CATEGORY || ')' || TITLE AS CATEGORY_TITLE,
  AUTHORID,
  TO_CHAR(STARTTIME, 'YYYYMMDDHH24MI') AS STARTTIME,
  STATUS  -- 모집중/모집완료/봉사완료/만료  *만료=시간 초과로 모집 종료
FROM VOLUNTEER 
where FLAG='o' -- r/o
order by VOLUNTEERID desc;
