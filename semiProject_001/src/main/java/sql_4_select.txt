SELECT VOLUNTEERID,
    AUTHORID,
     '(' || CATEGORY || ')' || TITLE AS CATEGORY_TITLE,
    CONTENT,
    PHONE,
    REGION,
    CATEGORY,
    CREATEDAT,
    TO_CHAR(STARTTIME, 'YYYYMMDDHH24MI') AS STARTTIME,
    TO_CHAR(ENDTIME, 'YYYYMMDDHH24MI') AS ENDTIME,
    STATUS,
    FLAG FROM VOLUNTEER
    order by VOLUNTEERID desc;
