SELECT VOLUNTEERID,
    AUTHORID,
     '(' || CATEGORY || ')' || TITLE AS CATEGORY_TITLE,
    CONTENT,
    PHONE,
    REGION,
    CATEGORY,
    TO_CHAR(CREATEDAT, 'YYYYMMDDHH24MI') AS CREATEDAT,
    TO_CHAR(STARTTIME, 'YYYYMMDDHH24MI') AS STARTTIME,
    TO_CHAR(ENDTIME, 'YYYYMMDDHH24MI') AS ENDTIME,
    STATUS,
    FLAG FROM VOLUNTEER
    WHERE flag = 'o' or 1010='1'
    order by VOLUNTEERID desc;

SELECT VOLUNTEERID, AUTHORID, '(' || CATEGORY || ')' || TITLE AS CAT_TITLE,

    REGION, STATUS, FLAG 
    FROM VOLUNTEER
